1.2 A MC for Brazilian Portuguese

1.2.1 Read and prepare words 

words <- read_lines('/usr/share/dict/brazilian') %>% 
  str_to_lower('pt-br')

A data frame, for easy search:

df <- tibble(word = words)

Characters that will be used to mark beginning and end are safe:

df %>% 
  filter(
    str_detect(word, '\\^') ||
    str_detect(word, '\\$')
  )

A list of character vectors, with added ‘^’ at the beginning and added double ‘$’ at the end of each:

words_to_vectors <- function(alist) (
  paste0('^', alist, '$$') %>% 
  strsplit('', fixed = TRUE)
)

chars <- words %>% words_to_vectors()

For a handful of words:

some_words <- chars[sample(1:length(chars), 10)]

1.2.3 Preprocessing with awk

This will read the dictionary and produce an R script to build a list:


BEGIN { print("wordlist <- list(") > "wordlist.R" }

{
    accented["á"] = "a´" 
    accented["à"] = "a`" 
    accented["â"] = "a^" 
    accented["ã"] = "a~" 
    accented["ç"] = "c," 
    accented["é"] = "e´" 
    accented["ê"] = "e^" 
    accented["í"] = "i´" 
    accented["ó"] = "o´" 
    accented["ô"] = "o^" 
    accented["õ"] = "o~" 
    accented["ú"] = "u´" 
    accented["ü"] = "u:" 

    word = tolower($0)

    # Split the word into array of chars
    split(word, vector, "")

    # Replace accented chars
    for (i in vector) {
        new = accented[vector[i]]
        if (new != "")
            vector[i] = new
    }

    # Output vector in R notation
    printf("  c('^', ") > "wordlist.R" 
    for (i in vector) printf("'%s', ", vector[i]) > "wordlist.R" 
    printf("'$', '$'),\n") > "wordlist.R" 
}

# Insert this NA just because of the last comma above,
# then remove it
END {
    print("  NA\n)\n") > "wordlist.R" 
    print("wordlist <- wordlist[-length(wordlist)]") > "wordlist.R" 
}
